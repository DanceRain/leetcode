# 翻转某一个区间内的链表

```c++
#include <iostream>

 using namespace std;

 struct ListNode {
    int val;
    struct ListNode *next = nullptr;
 };


ListNode* reverseBetween(ListNode* head, int m, int n)
{
    auto* pRes = new ListNode;
    pRes->next = head;
    auto* pPre = pRes;
    auto* pCur = head;
    for(int i = 1; i < m; ++i)
    {
        pPre = pCur;
        pCur = pCur->next;
    }
    for(int i = m; i < n; ++i)
    {
        auto* pTmp = pCur->next;
        //头插法
        pCur->next = pTmp->next;
        pTmp->next = pPre->next;
        pPre->next = pTmp;
    }
    return pRes->next;
}

ListNode* createList(int num)
{
    if(0 == num)
    {
        return nullptr;
    }
    auto* pHead = new ListNode;
    auto* pTmp = pHead;
    pHead->val = 1;
    while(--num)
    {
        pTmp->next = new ListNode;
        pTmp->next->val = pTmp->val + 1;
        pTmp = pTmp->next;
    }
    return pHead;
}

void printList(ListNode* pHead)
{
    auto* pTmp = pHead;
    while(pTmp != nullptr)
    {
        cout << pTmp->val << endl;
        pTmp = pTmp->next;
    }
}
int main()
{
    auto* pHead = createList(5);

    auto* p = reverseBetween(pHead, 1 ,3);
    while(p != nullptr)
    {
        cout << p->val << endl;
        p = p->next;
    }

    return 0;
}
```
